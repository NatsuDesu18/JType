    <p class="text-xl mb-6">Practice your Japanese typing skills with English-to-Japanese words or listen to the pronunciation.</p>

    <button id="startButton" class="bg-blue-600 text-white text-lg px-6 py-3 rounded-lg hover:bg-blue-700 transition">Get Started</button>

  </header>


  <!-- Typing Practice Section -->

  <main id="mainSection" class="flex flex-col items-center justify-center mt-16 hidden">

    <div id="englishWord" class="text-3xl font-semibold mb-4"></div>

    <audio id="audio" class="mb-4" controls>

      <source id="audioSource" type="audio/mpeg">

      Your browser does not support the audio element.

    </audio>

    <input type="text" id="userInput" placeholder="Type the Japanese word..." class="text-lg p-3 rounded-lg mb-4 w-1/2" />

    <div id="feedback" class="text-lg mt-4"></div>

    <button id="nextWordButton" class="mt-4 bg-blue-600 text-white text-lg px-6 py-3 rounded-lg hover:bg-blue-700 transition hidden">Next Word</button>

  </main>


  <footer class="text-center mt-12 text-sm text-gray-500">© 2025 Learn Japanese Typing. All rights reserved.</footer>


  <script>

    const words = [

      { english: "cat", japanese: "ねこ", audio: "https://www.example.com/audio/cat.mp3" },

      { english: "dog", japanese: "いぬ", audio: "https://www.example.com/audio/dog.mp3" },

      { english: "apple", japanese: "りんご", audio: "https://www.example.com/audio/apple.mp3" }

    ];


    let currentWordIndex = 0;


    document.getElementById('startButton').addEventListener('click', function() {

      document.getElementById('startButton').classList.add('hidden');

      document.getElementById('mainSection').classList.remove('hidden');

      showWord();

    });


    document.getElementById('userInput').addEventListener('input', function() {

      const userText = this.value.trim();

      const correctAnswer = words[currentWordIndex].japanese;

      const feedback = document.getElementById('feedback');


      if (userText === correctAnswer) {

        feedback.textContent = 'Correct! Well done.';

        feedback.classList.add('text-green-600');

        feedback.classList.remove('text-red-600');

        document.getElementById('nextWordButton').classList.remove('hidden');

      } else {

        feedback.textContent = 'Try again!';

        feedback.classList.add('text-red-600');

        feedback.classList.remove('text-green-600');

      }

    });


    document.getElementById('nextWordButton').addEventListener('click', function() {

      currentWordIndex = (currentWordIndex + 1) % words.length;

      showWord();

      document.getElementById('nextWordButton').classList.add('hidden');

      document.getElementById('userInput').value = '';

      document.getElementById('feedback').textContent = '';

    });


    function showWord() {

      const word = words[currentWordIndex];

      document.getElementById('englishWord').textContent = word.english;

      document.getElementById('audioSource').src = word.audio;

      document.getElementById('audio').load();

      document.getElementById('audio').play();

    }

  </script>
